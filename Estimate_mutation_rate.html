density2d_max <- function(x,y,n=100) {
  xx <- kde2d(x,y,n=n)
  p <- which.max(xx$z)
  n <- nrow(xx$z)
  iy <- floor((p-1)/n) + 1
  ix <- p - (iy-1) * n
  list(x = xx$x[ix],y=xx$y[iy])
}
